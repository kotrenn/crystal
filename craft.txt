hp_ugrade:
  label "HP Upgrade"
  input :Crystal src xN
  code {
    total = @sum(src$color)
    result = @min(total)
    @add_hp(result)
  }

mp_upgrade:
  label "Mana Upgrade"
  input :Crystal src xN
  code {
    total = @sum(src$color)
    @add_mana(total)
  }

mp_gen_upgrade:
  label "Mana Generation Upgrade"
  input :Crystal src xN
  req @geq(N, 10)
  code {
    total = @sum(src$color)
    total = @divide(total, 10)
    @add_mana_gen(total)
  }

pipe_upgrade:
  label "Randomize Connections"
  cost :Crystal x2
  input :Crystal src
  code {
    src.pipes = @random_pipes(src.pipes)
  }

color_randomize:
  label "Randomize Color"
  cost :Crystal x4
  input :Crystal src
  output :Crystal src x1
  code {
    src.color = @random_color()
  }

add_color:
  label "Color Addition"
  cost :Crystal x5
  input :Crystal lhs :crystal rhs 
  output :Crystal lhs x1
  code {
    dst.color = @add(lhs.color, rhs.color)
  }

att_upgrade:
  label "Attribute Upgrade"
  cost :Crystal x8
  input :Crystal src
  output :Crystal src
  code {
    att = @random_attribute(src.atts)
    att.val = @add(att.val, @randint(1, 3))
    @set_att(src.atts, att)
  }

decorrupt:
  label "Remove Corruption"
  cost :Crystal x20
  input :Spell src
  output :Spell src
  req @has_corruption(src)
  code {
    @remove_corruption(src)
  }

small_spell:
  label "Create Small Spell"
  cost :Crystal x20
  output :Spell dst
  code {
    dst = @Spell(2)
    @add_corruption(dst, 2)
  }

medium_spell:
  label "Create Medium Spell"
  cost :Crystal x20
  input :Spell src
  output :Spell dst :Crystal crystals xN
  req @eq(src.size, 2)
  code {
    crystals = @spell_crystals(src)
    dst = @Spell(3)
    @add_corruption(dst, 4)
  }

large_spell:
  label "Create Large Spell"
  cost :Crystal x20
  input :Spell src
  output :Spell dst :Crystal crystal xN
  req @eq(src.size, 3)
  code {
    crystals = @spell_crystals(src)
    dst = @Spell(4)
    @add_corruption(dst, 6)
  }
