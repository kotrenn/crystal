hp_upgrade:
  label "HP Upgrade"
  input :Crystal src xN
  code {
    total = @sum(src$color, @Color(0, 0, 0))
    result = @min(total)
    @add_hp(result)
  }

mp_upgrade:
  label "Mana Upgrade"
  input :Crystal src xN
  code {
    total = @sum(src$color, @Color(0, 0, 0))
    @add_mana(total)
  }

mp_gen_upgrade:
  label "Mana Generation Upgrade"
  input :Crystal src xN
  req {
    @geq(N, 10)
  }
  code {
    total = @sum(src$color, @Color(0, 0, 0))
    total = @divide(total, 10)
    @add_mana_gen(total)
  }

pipe_upgrade:
  label "Randomize Connections"
  cost :Crystal x2
  input :Crystal src
  output :Crystal src
  code {
    src.pipes = @random_pipes(src.pipes)
  }

color_randomize:
  label "Randomize Color"
  cost :Crystal x4
  input :Crystal src
  output :Crystal src
  code {
    src.color = @random_color()
  }

add_color:
  label "Color Addition"
  cost :Crystal x5
  input :Crystal lhs :Crystal rhs 
  output :Crystal lhs
  code {
    lhs.color = @add(lhs.color, rhs.color)
  }

att_upgrade:
  label "Attribute Upgrade"
  cost :Crystal x12
  input :Crystal src
  output :Crystal src
  code {
    att = @random_attribute(src.atts)
    val = @get_att(src.atts, att)
    new_val = @add(val, @randint(1, 3))
    @set_att(src.atts, att, new_val)
  }

mana_cost:
  label "Reduce Mana Cost"
  cost :Crystal cost x20
  input :Crystal src
  output :Crystal src
  req {
    @lt(@Color(0, 0, 0), src.atts['mana'])
    @leq(src.color, cost$atts['mana'])
  }
  code {
    r = @randint(0, 1)
    g = @randint(0, 1)
    b = @randint(0, 1)
    diff = @Color(r, g, b)
    src.color = @sub(src.color, diff)
  }

decorrupt:
  label "Remove Corruption"
  cost :Crystal x20
  input :Spell src
  output :Spell src
  req {
    @has_corruption(src)
  }
  code {
    @remove_corruption(src)
  }

small_spell:
  label "Create Small Spell"
  cost :Crystal x20
  output :Spell dst
  code {
    dst = @Spell(2)
    @add_corruption(dst, 2)
  }

medium_spell:
  label "Create Medium Spell"
  cost :Crystal x20
  input :Spell src
  output :Spell dst :Crystal crystals xN
  req {
    @eq(src.size, 2)
  }
  code {
    crystals = @spell_crystals(src)
    dst = @Spell(3)
    @add_corruption(dst, 4)
  }
